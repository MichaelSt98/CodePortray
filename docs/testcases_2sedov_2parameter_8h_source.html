279// Basic physical model, choose one of the following:
280// SOLID solves continuum mechanics with material strength, and stress tensor \sigma^{\alpha \beta} = -p \delta^{\alpha \beta} + S^{\alpha \beta}
281// HYDRO solves only the Euler equation, and there is only (scalar) pressure
282#define SOLID 1
283#define HYDRO 0
284// set additionally p to 0 if p < 0
285#define REAL_HYDRO 0
286
287// add additional point masses to the simulation, read from file <filename>.mass
288// format is location velocities mass r_min r_max, where location and velocities are vectors with size DIM and
289// r_min/r_max are min/max distances of sph particles to the bodies before they are taken out of the simulation
290#define GRAVITATING_POINT_MASSES 0
291
292// sink particles (set point masses to be sink particles)
293#define PARTICLE_ACCRETION 0 // check if particle is bound to one of the sink particles (crossed the accretion radius, rmin); if also UPDATE_SINK_VALUES 1: particle is accreted and ignored afterwards, else: continues orbiting without being accreted
294#define UPDATE_SINK_VALUES 0 // add to sink the quantities of the accreted particle: mass, velocity and COM
295
296// integrate the energy equation
297// when setting up a SOLID simulation with Tillotson or ANEOS, it must be set to 1
298#define INTEGRATE_ENERGY 0
299
300// integrate the continuity equation
301// if set to 0, the density will be calculated using the standard SPH sum \sum_i m_j W_ij
302#define INTEGRATE_DENSITY 1
303
304// adds viscosity to the Euler equation
305#define NAVIER_STOKES 0
306// choose between two different viscosity models
307#define SHAKURA_SUNYAEV_ALPHA 0
308#define CONSTANT_KINEMATIC_VISCOSITY 0
309// artificial bulk viscosity according to Schaefer et al. (2004)
310#define KLEY_VISCOSITY 0
311
312// This is the damage model following Benz & Asphaug (1995). Set FRAGMENTATION to activate it.
313// The damage acts always on pressure, but only on deviator stresses if DAMAGE_ACTS_ON_S is
314// activated too, which is an important switch depending on the plasticity model (see comments there).
315// Note: The damage model needs distribution of activation thresholds in the input file.
316#define FRAGMENTATION 0
317#define DAMAGE_ACTS_ON_S 0
318
319// Choose the SPH representation to solve the momentum and energy equation:
320// SPH_EQU_VERSION 1: original version with HYDRO dv_a/dt ~ - (p_a/rho_a**2 + p_b/rho_b**2) \nabla_a W_ab
321// SOLID dv_a/dt ~ (sigma_a/rho_a**2 + sigma_b/rho_b**2) \nabla_a W_ab
322// SPH_EQU_VERSION 2: slighty different version with
323// HYDRO dv_a/dt ~ - (p_a+p_b)/(rho_a*rho_b) \nabla_a W_ab
324// SOLID dv_a/dt ~ (sigma_a+sigma_b)/(rho_a*rho_b) \nabla_a W_ab
325// If you do not know what to do, choose SPH_EQU_VERSION 1.
326#define SPH_EQU_VERSION 1
327
328// for the tensile instability fix
329// you do not need this
330#define ARTIFICIAL_STRESS 0
331
332// standard SPH alpha/beta viscosity
333#define ARTIFICIAL_VISCOSITY 1
334// Balsara switch: lowers the artificial viscosity in regions without shocks
335#define BALSARA_SWITCH 0
336
337// INVISCID SPH (see Cullen & Dehnen paper)
338#define INVISCID_SPH 0
339
340// consistency switches
341// for zeroth order consistency
342#define SHEPARD_CORRECTION 0
343// for linear consistency
344// add tensorial correction tensor to dSdt calculation -> better conservation of angular momentum
345#define TENSORIAL_CORRECTION 1
346
347// Available plastic flow conditions:
348// (if you do not know what this is, choose (1) or nothing)
349// (1) Simple von Mises plasticity with a constant yield strength:
350#define VON_MISES_PLASTICITY 0
351// (2) Drucker-Prager (DP) yield criterion -> yield strength is given by the condition \sqrt(J_2) + A * I_1 + B = 0
352// with I_1: first invariant of stress tensor
353// J_2: second invariant of stress tensor
354// A, B: DP constants, which are calculated from angle of internal friction and cohesion
355#define DRUCKER_PRAGER_PLASTICITY 0
356// (3) Mohr-Coulomb (MC) yield criterion
357// -> yield strength is given by yield_stress = tan(friction_angle) \times pressure + cohesion
358#define MOHR_COULOMB_PLASTICITY 0
359// Note: DP and MC are intended for granular-like materials, therefore the yield strength simply decreases (linearly) to zero for p<0.
360// Note: For DP and MC you can additionally choose (1) to impose an upper limit for the yield stress.
361// (4) Pressure dependent yield strength following Collins et al. (2004) and the implementation in Jutzi (2015)
362// -> yield strength is different for damaged (Y_d) and intact material (Y_i), and averaged mean (Y) in between:
363// Y_i = cohesion + \mu P / (1 + \mu P / (yield_stress - cohesion) )
364// where *cohesion* is the yield strength for P=0 and *yield_stress* the asymptotic limit for P=\infty
365// \mu is the coefficient of internal friction (= tan(friction_angle))
366// Y_d = cohesion_damaged + \mu_d \times P
367// where \mu_d is the coefficient of friction of the *damaged* material
368// Y = (1-damage)*Y_i + damage*Y_d
369// Y is limited to <= Y_i
370// Note: If FRAGMENTATION is not activated only Y_i is used.
371// DAMAGE_ACTS_ON_S is not allowed for this model, since the limiting of S already depends on damage.
372// If you want to additionally model the influence of some (single) melt energy on the yield strength, then activate
373// COLLINS_PLASTICITY_INCLUDE_MELT_ENERGY, which adds a factor (1-e/e_melt) to the yield strength.
374#define COLLINS_PLASTICITY 0
375#define COLLINS_PLASTICITY_INCLUDE_MELT_ENERGY 0
376// (5) Simplified version of the Collins et al. (2004) model, which uses only the
377// strength representation for intact material (Y_i), irrespective of damage.
378// Unlike in (4), Y decreases to zero (following a linear yield strength curve) for p<0.
379// In addition, negative pressures are limited to the pressure corresponding to
380// yield strength = 0 (i.e., are set to this value when they get more negative).
381#define COLLINS_PLASTICITY_SIMPLE 0
382// Note: The deviator stress tensor is additionally reduced by FRAGMENTATION (i.e., damage) only if
383// DAMAGE_ACTS_ON_S is set. For most plasticity models it depends on the use case whether this
384// is desired, only for COLLINS_PLASTICITY it is not reasonable (and therefore not allowed).
385
386// model regolith as viscous fluid -> experimental setup, only for powerusers

391#define JC_PLASTICITY 0
