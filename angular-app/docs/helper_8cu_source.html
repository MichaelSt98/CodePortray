101
102 // device wide reduction
103 void *d_temp_storage = NULL;
104 size_t temp_storage_bytes = 0;
105 switch (reductionType) {
106 case Reduction::min: {

108 // Allocate temporary storage
109 cuda::malloc(d_temp_storage, temp_storage_bytes);
110 // Run max-reduction
111 cub::DeviceReduce::Min(d_temp_storage, temp_storage_bytes, d_sml, d_aggregate, n);
112 } break;
113 case Reduction::max: {
114 cub::DeviceReduce::Max(d_temp_storage, temp_storage_bytes, d_sml, d_aggregate, n);
115 // Allocate temporary storage
116 cuda::malloc(d_temp_storage, temp_storage_bytes);
117 // Run max-reduction
118 cub::DeviceReduce::Max(d_temp_storage, temp_storage_bytes, d_sml, d_aggregate, n);
119 } break;
120 case Reduction::sum: {
121 cub::DeviceReduce::Sum(d_temp_storage, temp_storage_bytes, d_sml, d_aggregate, n);
122 // Allocate temporary storage
123 cuda::malloc(d_temp_storage, temp_storage_bytes);
124 // Run max-reduction
125 cub::DeviceReduce::Sum(d_temp_storage, temp_storage_bytes, d_sml, d_aggregate, n);
126 } break;
127 default: {
128 Logger(ERROR) << "Reduction type not available!";
129 }
130 }
131
132 T reduction;
133 gpuErrorcheck(cudaMemcpy(&reduction, d_aggregate, sizeof(T), cudaMemcpyDeviceToHost));
134 Logger(INFO) << "reduction = " << reduction;

136 switch (reductionType) {
137 case Reduction::min: {
138 // interprocess reduction
139 boost::mpi::communicator comm;
140 all_reduce(comm, boost::mpi::inplace_t<T *>(&reduction), 1, boost::mpi::minimum<T>());
141 } break;
142 case Reduction::max: {
143 // interprocess reduction
144 boost::mpi::communicator comm;
145 all_reduce(comm, boost::mpi::inplace_t<T *>(&reduction), 1, boost::mpi::maximum<T>());
146 } break;
147 case Reduction::sum: {
148 // interprocess reduction
149 boost::mpi::communicator comm;
150 all_reduce(comm, boost::mpi::inplace_t<T *>(&reduction), 1, std::plus<T>());
151 } break;
152 default: {
153 Logger(ERROR) << "Reduction type not available!";
154 }
155 }
156 Logger(INFO) << "globalized reduction = " << reduction;

158 cuda::free(d_temp_storage);
