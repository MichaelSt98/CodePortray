132 // track single values
133 template<typename T>
134 void createValueDataSet(const std::string& path, int steps, std::size_t maxSteps=HighFive::DataSpace::UNLIMITED){
135 HighFive::DataSetCreateProps props;
136 props.add(HighFive::Chunking(std::vector<hsize_t>{1, 1}));
137 dataSets[path] = h5file.createDataSet<T>(path, HighFive::DataSpace({std::size_t(steps), std::size_t(numProcs)},
138 {maxSteps, std::size_t(numProcs)}), props);
139 totalSteps[path] = steps;
140 currentSteps[path] = 0;
141 }
142
143 template<typename T>
144 void value2file(const std::string& path, T value) {
145
146 if (!disabled) {
147 if (currentSteps[path] >= totalSteps[path]) {
148 totalSteps[path] += 1;
149 //Logger(INFO) << "resizing: " << totalSteps[path];
150 dataSets[path].resize({std::size_t(totalSteps[path]), std::size_t(numProcs)});
151 }
152 dataSets[path].select({std::size_t(currentSteps[path]), std::size_t(myRank)}, {1, 1}).write(value);
153 }
154 currentSteps[path] += 1;
155 }
156
157 // track vector values
158 template<typename T>
159 void createVectorDataSet(const std::string& path, int steps, std::size_t size,
160 std::size_t maxSteps=HighFive::DataSpace::UNLIMITED){
161 HighFive::DataSetCreateProps props;
162 props.add(HighFive::Chunking(std::vector<hsize_t>{1, 1, size}));
163 dataSets[path] = h5file.createDataSet<T>(path, HighFive::DataSpace({std::size_t(steps),
164 std::size_t(numProcs), size},
165 {maxSteps, std::size_t(numProcs), size}),
166 props);
167 vectorSizes[path] = size;
168 currentSteps[path] = 0;
169 }
170
171 template<typename T>
172 void vector2file(const std::string& path, std::vector<T> data){
173 if (!disabled) {
174 if (currentSteps[path] >= totalSteps[path]) {
175 totalSteps[path] += 1;
176 //Logger(INFO) << "resizing: " << totalSteps[path];
177 dataSets[path].resize({std::size_t(totalSteps[path]), std::size_t(numProcs), vectorSizes[path]});
178 }
179 dataSets[path].select({std::size_t(currentSteps[path]), std::size_t(myRank), 0},
180 {1, 1, vectorSizes[path]}).write(data);
181 }
182 currentSteps[path] += 1;
183 }
184
185 template<typename T>
186 void vector2file(const std::string& path, T *data) {
187 if (!disabled) {
188 std::vector<T> dataVector;
189 for (int i=0; i<vectorSizes[path]; i++) {
190 dataVector.push_back(data[i]);
191 }
192 vector2file(path, dataVector);
193 }
194 }
