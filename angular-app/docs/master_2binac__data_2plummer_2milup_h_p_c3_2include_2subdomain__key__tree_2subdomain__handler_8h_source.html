137
138 boost::mpi::communicator comm;
139
140 std::vector <boost::mpi::request> reqParticles;
141 std::vector <boost::mpi::status> statParticles;
142
143 for (int proc = 0; proc < subDomainKeyTree->numProcesses; proc++) {
144 if (proc != subDomainKeyTree->rank) {
145 reqParticles.push_back(comm.isend(proc, 17, &toSend[proc], 1));
146 statParticles.push_back(comm.recv(proc, 17, &toReceive[proc], 1));
147 }
148 }
149
150 boost::mpi::wait_all(reqParticles.begin(), reqParticles.end());
151 }
